```python
# dashboard/components/cards.py

from nicegui import ui

def kpi_card(title: str, value: int, color: str):
    colors = {
        'blue': 'bg-blue-900',
        'green': 'bg-green-900',
        'purple': 'bg-purple-900',
        'orange': 'bg-orange-900',
    }

    with ui.card().classes(f'{colors[color]} p-4 min-w-[200px]'):
        ui.label(title).classes('text-gray-400 text-sm')
        ui.label(str(value)).classes('text-3xl font-bold text-white')
```
